## 2. 核心组件详解

### 2.1 Session Manager Service

**职责**: 管理用户调试会话的生命周期

**核心功能**:

```typescript
class SessionManager {
  // 创建新会话
  async createSession(userId: string, recordingId: string): Promise<Session> {
    // 1. 验证用户权限
    await this.authService.checkAccess(userId, recordingId);

    // 2. 生成会话 ID
    const sessionId = generateUUID();

    // 3. 从 S3 获取录制元数据
    const metadata = await this.s3.getRecordingMetadata(recordingId);

    // 4. 创建 Kubernetes Pod
    const pod = await this.k8s.createReplayEnginePod({
      sessionId,
      recordingId,
      resources: this.calculateResources(metadata),
      env: {
        REPLAY_MODE: 'REPLAYING',
        RECORDING_ID: recordingId,
        SESSION_ID: sessionId,
      }
    });

    // 5. 等待 Pod 就绪
    await this.waitForPodReady(pod, timeout: 30000);

    // 6. 建立 WebSocket 连接
    const wsUrl = await this.getWebSocketUrl(pod);

    // 7. 保存会话信息到 Redis
    await this.redis.setSession(sessionId, {
      userId,
      recordingId,
      podName: pod.name,
      wsUrl,
      createdAt: Date.now(),
      status: 'active'
    });

    return {
      sessionId,
      wsUrl,
      status: 'ready'
    };
  }

  // 销毁会话
  async destroySession(sessionId: string): Promise<void> {
    const session = await this.redis.getSession(sessionId);

    // 1. 删除 Kubernetes Pod
    await this.k8s.deletePod(session.podName);

    // 2. 清理 Redis 缓存
    await this.redis.deleteSession(sessionId);

    // 3. 记录会话统计
    await this.analytics.recordSessionEnd(sessionId);
  }

  // 计算资源需求
  calculateResources(metadata: RecordingMetadata): Resources {
    // 基于录制时长和复杂度计算
    const duration = metadata.duration;
    const complexity = metadata.complexity || 1;

    return {
      cpu: Math.min(2 + Math.floor(duration / 30), 8), // 最多 8 核
      memory: Math.min(4 + Math.floor(duration / 10), 16), // 最多 16GB
      ephemeralStorage: Math.min(10 + Math.floor(duration / 5), 50) // 最多 50GB
    };
  }
}
```

### 2.3 Replay Engine 启动流程

```typescript
class ReplayEngine {
  async start() {
    console.log('[ReplayEngine] Starting...');

    // 1. 初始化环境
    await this.initializeEnvironment();

    // 2. 从 S3 下载录制文件
    const recordingPath = await this.downloadRecording();

    // 3. 加载录制文件到内存
    await this.loadRecording(recordingPath);

    // 4. 构建执行点索引
    await this.buildExecutionPointIndex();

    // 5. 创建初始检查点
    await this.createInitialCheckpoint();

    // 6. 启动 CDP 服务器
    await this.startCDPServer();

    // 7. 通知 Session Manager 就绪
    await this.notifyReady();

    console.log('[ReplayEngine] Ready');
  }

  async downloadRecording(): Promise<string> {
    const recordingId = process.env.RECORDING_ID;
    const s3Key = `recordings/${recordingId}.dat`;
    const localPath = `/cache/${recordingId}.dat`;

    console.log(`[ReplayEngine] Downloading ${s3Key}...`);

    // 流式下载，支持断点续传
    await this.s3.downloadFile(s3Key, localPath, {
      onProgress: (progress) => {
        console.log(`[ReplayEngine] Download progress: ${progress}%`);
      }
    });

    return localPath;
  }

  async loadRecording(path: string): Promise<void> {
    console.log('[ReplayEngine] Loading recording...');

    // 调用 C++ 层的 RecordReplay API
    const success = RecordReplayAttach(path);

    if (!success) {
      throw new Error('Failed to load recording');
    }

    // 验证录制文件完整性
    const recordingId = RecordReplayGetRecordingId();
    if (recordingId !== process.env.RECORDING_ID) {
      throw new Error('Recording ID mismatch');
    }

    console.log('[ReplayEngine] Recording loaded');
  }

  async buildExecutionPointIndex(): Promise<void> {
    console.log('[ReplayEngine] Building execution point index...');

    // 扫描录制文件，构建执行点索引
    // 这样可以快速跳转到任意执行点
    const index = {
      totalPoints: 0,
      checkpoints: [],
      breakpoints: new Map(),
      sourceLocations: new Map()
    };

    // 遍历所有执行点
    let point = 0;
    while (true) {
      const info = RecordReplayGetExecutionPointInfo(point);
      if (!info) break;

      index.totalPoints++;

      // 每 N 个执行点创建一个检查点引用
      if (point % CHECKPOINT_INTERVAL === 0) {
        index.checkpoints.push({
          point,
          offset: info.fileOffset
        });
      }

      // 记录源码位置映射
      if (info.sourceLocation) {
        const key = `${info.scriptId}:${info.line}:${info.column}`;
        if (!index.sourceLocations.has(key)) {
          index.sourceLocations.set(key, []);
        }
        index.sourceLocations.get(key).push(point);
      }

      point++;
    }

    this.executionPointIndex = index;
    console.log(`[ReplayEngine] Index built: ${index.totalPoints} points`);
  }
}
```

```typescript
function handleSetBreakpoint(params) {
  const { location } = params;

  // 1. 查找执行点
  const key = `${location.scriptId}:${location.lineNumber}:0`;
  const points = this.executionPointIndex.sourceLocations.get(key);

  if (!points || points.length === 0) {
    return { error: 'No executable code at location' };
  }

  // 2. 创建断点
  const breakpointId = generateUUID();
  this.breakpoints.set(breakpointId, {
    location,
    points,
    enabled: true
  });

  // 3. 返回实际位置
  return {
    breakpointId,
    actualLocation: location
  };
}
```

### 3.3 执行到断点

```typescript
class ReplayEngine {
  async runToBreakpoint(targetPoint: number) {
    console.log(`[ReplayEngine] Running to point ${targetPoint}...`);

    // 1. 检查是否需要从检查点恢复
    if (targetPoint < this.currentPoint) {
      await this.restoreFromCheckpoint(targetPoint);
    }

    // 2. 执行到目标点
    while (this.currentPoint < targetPoint) {
      // 执行下一条指令
      await this.executeNextInstruction();

      this.currentPoint++;

      // 检查是否命中断点
      if (this.isBreakpoint(this.currentPoint)) {
        await this.handleBreakpointHit(this.currentPoint);
        break;
      }
    }

    console.log(`[ReplayEngine] Paused at point ${this.currentPoint}`);
  }

  async handleBreakpointHit(point: number) {
    // 1. 收集暂停状态
    const pauseData = {
      callFrames: this.getCallFrames(),
      reason: 'breakpoint',
      hitBreakpoints: this.getHitBreakpoints(point),
      data: {}
    };

    // 2. 发送 CDP 事件
    this.cdp.sendEvent('Debugger.paused', pauseData);

    // 3. 等待用户操作
    await this.waitForResume();
  }

  getCallFrames(): CallFrame[] {
    const frames = [];

    // 遍历调用栈
    let frameIndex = 0;
    while (true) {
      const frame = RecordReplayGetFrame(frameIndex);
      if (!frame) break;

      frames.push({
        callFrameId: `frame-${frameIndex}`,
        functionName: frame.functionName,
        location: {
          scriptId: frame.scriptId,
          lineNumber: frame.line,
          columnNumber: frame.column
        },
        scopeChain: this.getScopeChain(frameIndex),
        this: this.getThisObject(frameIndex)
      });

      frameIndex++;
    }

    return frames;
  }
}
```

## 4. 检查点机制

### 4.1 检查点创建

```typescript
class CheckpointManager {
  async createCheckpoint(point: number): Promise<Checkpoint> {
    console.log(`[Checkpoint] Creating at point ${point}...`);

    const checkpoint = {
      point,
      timestamp: Date.now(),

      // 内存快照
      memory: await this.captureMemorySnapshot(),

      // 调用栈
      callStack: this.captureCallStack(),

      // 全局状态
      globalState: this.captureGlobalState(),

      // DOM 状态
      domState: await this.captureDOMState(),

      // V8 堆快照
      heapSnapshot: await this.captureHeapSnapshot()
    };

    // 压缩并存储
    const compressed = await this.compress(checkpoint);
    await this.storage.saveCheckpoint(point, compressed);

    console.log(`[Checkpoint] Created (${compressed.size} bytes)`);

    return checkpoint;
  }

  async restoreCheckpoint(point: number): Promise<void> {
    console.log(`[Checkpoint] Restoring from point ${point}...`);

    // 1. 查找最近的检查点
    const checkpoint = await this.findNearestCheckpoint(point);

    if (!checkpoint) {
      // 从头开始
      await this.resetToBeginning();
      return;
    }

    // 2. 加载检查点数据
    const data = await this.storage.loadCheckpoint(checkpoint.point);
    const decompressed = await this.decompress(data);

    // 3. 恢复内存
    await this.restoreMemorySnapshot(decompressed.memory);

    // 4. 恢复调用栈
    await this.restoreCallStack(decompressed.callStack);

    // 5. 恢复全局状态
    await this.restoreGlobalState(decompressed.globalState);

    // 6. 恢复 DOM
    await this.restoreDOMState(decompressed.domState);

    // 7. 恢复 V8 堆
    await this.restoreHeapSnapshot(decompressed.heapSnapshot);

    this.currentPoint = checkpoint.point;

    console.log(`[Checkpoint] Restored to point ${checkpoint.point}`);
  }
}
```

### 4.2 检查点策略

```typescript
class CheckpointStrategy {
  // 自适应检查点间隔
  calculateCheckpointInterval(metadata: RecordingMetadata): number {
    const duration = metadata.duration;
    const complexity = metadata.complexity;

    // 短录制：更频繁的检查点
    if (duration < 10000) {
      return 1000; // 每 1 秒
    }

    // 长录制：较少的检查点
    if (duration > 60000) {
      return 5000; // 每 5 秒
    }

    // 中等录制
    return 2000; // 每 2 秒
  }

  // 预测性检查点
  async createPredictiveCheckpoints(userBehavior: UserBehavior) {
    // 基于用户行为预测可能访问的执行点
    const predictions = this.predictNextAccess(userBehavior);

    for (const point of predictions) {
      if (!this.hasCheckpoint(point)) {
        await this.createCheckpoint(point);
      }
    }
  }
}
```

### 5.2 缓存策略

```typescript
class CacheManager {
  // 多级缓存
  private l1Cache: Map<number, Checkpoint>; // 内存缓存
  private l2Cache: Redis; // Redis 缓存
  private l3Cache: S3; // S3 持久化

  async getCheckpoint(point: number): Promise<Checkpoint> {
    // L1: 内存
    if (this.l1Cache.has(point)) {
      return this.l1Cache.get(point);
    }

    // L2: Redis
    const cached = await this.l2Cache.get(`checkpoint:${point}`);
    if (cached) {
      const checkpoint = JSON.parse(cached);
      this.l1Cache.set(point, checkpoint);
      return checkpoint;
    }

    // L3: S3
    const data = await this.l3Cache.getObject(`checkpoints/${point}`);
    const checkpoint = await this.decompress(data);

    // 回填缓存
    this.l2Cache.set(`checkpoint:${point}`, JSON.stringify(checkpoint), 'EX', 3600);
    this.l1Cache.set(point, checkpoint);

    return checkpoint;
  }
}
```

```typescript
class AutoScaler {
  async scale() {
    const metrics = await this.getMetrics();

    // 1. 检查当前负载
    const currentLoad = metrics.activeSessions / metrics.availablePods;

    // 2. 决定扩缩容
    if (currentLoad > 0.8) {
      // 扩容
      await this.scaleUp();
    } else if (currentLoad < 0.3) {
      // 缩容
      await this.scaleDown();
    }
  }

  async scaleUp() {
    const currentReplicas = await this.k8s.getReplicaCount('replay-engine');
    const targetReplicas = Math.min(currentReplicas + 5, MAX_REPLICAS);

    await this.k8s.scale('replay-engine', targetReplicas);
  }
}
```

```typescript
      checkpointRestoreTime: this.getCheckpointRestoreTime(),
      executionPointsPerSecond: this.getExecutionPointsPerSecond(),

      // 错误指标
      errorRate: this.getErrorRate(),
      divergenceRate: this.getDivergenceRate()
    };
  }
```

## 8. 总结

### 8.1 核心技术栈

- **容器编排**: Kubernetes
- **存储**: S3 (录制文件) + Redis (缓存) + PostgreSQL (元数据)
- **通信**: WebSocket (CDP 协议)
- **浏览器**: Replay Chromium (改造版)
- **语言**: TypeScript (后端服务) + C++ (Replay Engine)

### 8.2 关键设计

1. **动态 Pod 创建**: 每个会话独立的 Replay Engine
2. **检查点机制**: 快速时间旅行
3. **预分析**: 提前构建索引和检查点
4. **多级缓存**: 优化性能
5. **自动扩缩容**: 应对负载变化
6. **资源回收**: 控制成本

### 8.3 挑战与解决方案

| 挑战 | 解决方案 |
|------|---------|
| 资源消耗大 | 动态创建/销毁 Pod，自动扩缩容 |
| 启动慢 | 预分析，预热检查点 |
| 跳转慢 | 检查点机制，增量重放 |
| 成本高 | 资源回收，空闲超时 |
| 并发高 | Kubernetes 水平扩展 |
