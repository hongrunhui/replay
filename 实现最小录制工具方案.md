# 基于 V8/Chromium 实现最小 JS 执行录制工具

## 方案对比

| 特性 | 方案 1: V8 引擎 | 方案 2: Chromium CDP |
|------|----------------|---------------------|
| 难度 | 高（需要 C++） | 中（主要是 JS/TS） |
| 完整性 | 完整（底层控制） | 有限（依赖 CDP） |
| 开发时间 | 长（数周） | 短（数天） |
| 浏览器环境 | 需要自己实现 | 原生支持 |
| 推荐度 | ⭐⭐⭐ | ⭐⭐⭐⭐⭐ |

**推荐**: 先从方案 2 开始，快速验证概念，再考虑方案 1 深度定制。

```cpp
#ifndef V8_RECORDER_RECORDER_H_
#define V8_RECORDER_RECORDER_H_

#include <fstream>
#include <vector>
#include "include/v8.h"

namespace v8 {
namespace internal {

// 执行点记录
struct ExecutionPoint {
  uint64_t id;
  uint64_t timestamp;
  int bytecode_offset;
  const char* function_name;
  int line_number;
  int column_number;
};

// 非确定性数据记录
struct NonDeterministicData {
  enum Type {
    RANDOM,
    TIME,
    EXTERNAL_CALL
  };

  Type type;
  uint64_t execution_point;
  std::vector<uint8_t> data;
};

class Recorder {
 public:
  static Recorder* GetInstance();

  void StartRecording(const char* output_file);
  void StopRecording();
  bool IsRecording() const { return is_recording_; }

  // 记录执行点
  void RecordExecutionPoint(
      int bytecode_offset,
      const char* function_name,
      int line_number,
      int column_number);

  // 记录非确定性数据
  void RecordRandom(double value);
  void RecordTime(double value);
  void RecordExternalCall(const char* name, const uint8_t* data, size_t len);

 private:
  Recorder() : is_recording_(false), next_point_id_(0) {}

  bool is_recording_;
  uint64_t next_point_id_;
  std::ofstream output_file_;
  std::vector<ExecutionPoint> execution_points_;
  std::vector<NonDeterministicData> non_deterministic_data_;

  void Flush();
};

} // namespace internal
} // namespace v8

#endif  // V8_RECORDER_RECORDER_H_
```

创建 `src/recorder/recorder.cc`:

```cpp
#include "src/recorder/recorder.h"
#include <chrono>

namespace v8 {
namespace internal {

Recorder* Recorder::GetInstance() {
  static Recorder instance;
  return &instance;
}

void Recorder::StartRecording(const char* output_file) {
  output_file_.open(output_file, std::ios::binary);
  is_recording_ = true;
  next_point_id_ = 0;

  // 写入文件头
  const char* magic = "V8REC001";
  output_file_.write(magic, 8);
}

void Recorder::StopRecording() {
  if (!is_recording_) return;

  Flush();
  output_file_.close();
  is_recording_ = false;
}

void Recorder::RecordExecutionPoint(
    int bytecode_offset,
    const char* function_name,
    int line_number,
    int column_number) {

  if (!is_recording_) return;

  ExecutionPoint point;
  point.id = next_point_id_++;
  point.timestamp = std::chrono::steady_clock::now().time_since_epoch().count();
  point.bytecode_offset = bytecode_offset;
  point.function_name = function_name;
  point.line_number = line_number;
  point.column_number = column_number;

  execution_points_.push_back(point);

  // 每 1000 个执行点刷新一次
  if (execution_points_.size() >= 1000) {
    Flush();
  }
}

void Recorder::RecordRandom(double value) {
  if (!is_recording_) return;

  NonDeterministicData data;
  data.type = NonDeterministicData::RANDOM;
  data.execution_point = next_point_id_ - 1;
  data.data.resize(sizeof(double));
  memcpy(data.data.data(), &value, sizeof(double));

  non_deterministic_data_.push_back(data);
}

void Recorder::RecordTime(double value) {
  if (!is_recording_) return;

  NonDeterministicData data;
  data.type = NonDeterministicData::TIME;
  data.execution_point = next_point_id_ - 1;
  data.data.resize(sizeof(double));
  memcpy(data.data.data(), &value, sizeof(double));

  non_deterministic_data_.push_back(data);
}

void Recorder::Flush() {
  // 写入执行点
  uint32_t count = execution_points_.size();
  output_file_.write(reinterpret_cast<char*>(&count), sizeof(count));

  for (const auto& point : execution_points_) {
    output_file_.write(reinterpret_cast<const char*>(&point.id), sizeof(point.id));
    output_file_.write(reinterpret_cast<const char*>(&point.timestamp), sizeof(point.timestamp));
    output_file_.write(reinterpret_cast<const char*>(&point.bytecode_offset), sizeof(point.bytecode_offset));

    uint32_t name_len = strlen(point.function_name);
    output_file_.write(reinterpret_cast<char*>(&name_len), sizeof(name_len));
    output_file_.write(point.function_name, name_len);

    output_file_.write(reinterpret_cast<const char*>(&point.line_number), sizeof(point.line_number));
    output_file_.write(reinterpret_cast<const char*>(&point.column_number), sizeof(point.column_number));
  }

  execution_points_.clear();

  // 写入非确定性数据
  count = non_deterministic_data_.size();
  output_file_.write(reinterpret_cast<char*>(&count), sizeof(count));

  for (const auto& data : non_deterministic_data_) {
    output_file_.write(reinterpret_cast<const char*>(&data.type), sizeof(data.type));
    output_file_.write(reinterpret_cast<const char*>(&data.execution_point), sizeof(data.execution_point));

    uint32_t data_len = data.data.size();
    output_file_.write(reinterpret_cast<char*>(&data_len), sizeof(data_len));
    output_file_.write(reinterpret_cast<const char*>(data.data.data()), data_len);
  }

  non_deterministic_data_.clear();
}

} // namespace internal
} // namespace v8
```

```cpp
// example.cc
#include "include/v8.h"
#include <iostream>

int main(int argc, char* argv[]) {
  // 初始化 V8
  v8::V8::InitializeICUDefaultLocation(argv[0]);
  v8::V8::InitializeExternalStartupData(argv[0]);
  std::unique_ptr<v8::Platform> platform = v8::platform::NewDefaultPlatform();
  v8::V8::InitializePlatform(platform.get());
  v8::V8::Initialize();

  // 创建 Isolate
  v8::Isolate::CreateParams create_params;
  create_params.array_buffer_allocator =
      v8::ArrayBuffer::Allocator::NewDefaultAllocator();
  v8::Isolate* isolate = v8::Isolate::New(create_params);

  {
    v8::Isolate::Scope isolate_scope(isolate);
    v8::HandleScope handle_scope(isolate);
    v8::Local<v8::Context> context = v8::Context::New(isolate);
    v8::Context::Scope context_scope(context);

    // 开始录制
    v8::V8::StartRecording("recording.dat");

    // 执行 JavaScript
    const char* source = R"(
      function fibonacci(n) {
        if (n <= 1) return n;
        return fibonacci(n - 1) + fibonacci(n - 2);
      }

      console.log('Result:', fibonacci(10));
      console.log('Random:', Math.random());
      console.log('Time:', Date.now());
    )";

    v8::Local<v8::String> source_string =
        v8::String::NewFromUtf8(isolate, source).ToLocalChecked();
    v8::Local<v8::Script> script =
        v8::Script::Compile(context, source_string).ToLocalChecked();
    script->Run(context).ToLocalChecked();

    // 停止录制
    v8::V8::StopRecording();
  }

  // 清理
  isolate->Dispose();
  v8::V8::Dispose();
  v8::V8::DisposePlatform();
  delete create_params.array_buffer_allocator;

  return 0;
}
```

创建 `src/recorder.ts`:

```typescript
import puppeteer, { Browser, Page, CDPSession } from 'puppeteer';
import * as fs from 'fs';

interface ExecutionPoint {
  id: number;
  timestamp: number;
  scriptId: string;
  lineNumber: number;
  columnNumber: number;
  functionName?: string;
  callFrames?: any[];
}

interface NonDeterministicData {
  type: 'random' | 'time' | 'network';
  executionPoint: number;
  data: any;
}

export class JSRecorder {
  private browser?: Browser;
  private page?: Page;
  private cdp?: CDPSession;
  private recording: boolean = false;
  private executionPoints: ExecutionPoint[] = [];
  private nonDeterministicData: NonDeterministicData[] = [];
  private nextPointId: number = 0;
  private outputFile?: string;

  async start(url: string, outputFile: string) {
    this.outputFile = outputFile;
    this.recording = true;

    // 启动 Chromium
    this.browser = await puppeteer.launch({
      headless: false,
      devtools: true,
      args: ['--enable-precise-memory-info']
    });

    this.page = await this.browser.newPage();
    this.cdp = await this.page.target().createCDPSession();

    // 启用必要的 CDP 域
    await this.cdp.send('Debugger.enable');
    await this.cdp.send('Runtime.enable');
    await this.cdp.send('Profiler.enable');

    // 设置断点在每个语句
    await this.cdp.send('Debugger.setPauseOnExceptions', { state: 'none' });
    await this.cdp.send('Debugger.setBreakpointsActive', { active: true });

    // 监听调试器事件
    this.setupEventListeners();

    // 注入录制脚本
    await this.injectRecordingScript();

    // 导航到目标 URL
    await this.page.goto(url);
  }

  private setupEventListeners() {
    if (!this.cdp) return;

    // 监听脚本解析
    this.cdp.on('Debugger.scriptParsed', (params) => {
      console.log(`Script parsed: ${params.scriptId} - ${params.url}`);
    });

    // 监听断点命中
    this.cdp.on('Debugger.paused', async (params) => {
      if (!this.recording) return;

      // 记录执行点
      const point: ExecutionPoint = {
        id: this.nextPointId++,
        timestamp: Date.now(),
        scriptId: params.callFrames[0].location.scriptId,
        lineNumber: params.callFrames[0].location.lineNumber,
        columnNumber: params.callFrames[0].location.columnNumber || 0,
        functionName: params.callFrames[0].functionName,
        callFrames: params.callFrames
      };

      this.executionPoints.push(point);

      // 继续执行
      await this.cdp!.send('Debugger.resume');
    });

    // 监听控制台输出
    this.cdp.on('Runtime.consoleAPICalled', (params) => {
      console.log('Console:', params.type, params.args);
    });
  }

  private async injectRecordingScript() {
    if (!this.page) return;

    // 注入脚本拦截非确定性函数
    await this.page.evaluateOnNewDocument(() => {
      // 保存原始函数
      const originalRandom = Math.random;
      const originalDateNow = Date.now;
      const originalFetch = window.fetch;

      // 录制数据存储
      (window as any).__recordingData = {
        random: [],
        time: [],
        network: []
      };

      // 拦截 Math.random
      Math.random = function() {
        const value = originalRandom();
        (window as any).__recordingData.random.push({
          timestamp: Date.now(),
          value
        });
        return value;
      };

      // 拦截 Date.now
      Date.now = function() {
        const value = originalDateNow();
        (window as any).__recordingData.time.push({
          timestamp: value,
          value
        });
        return value;
      };

      // 拦截 fetch
      window.fetch = function(...args) {
        const startTime = Date.now();
        return originalFetch.apply(this, args).then(response => {
          (window as any).__recordingData.network.push({
            timestamp: startTime,
            url: args[0],
            status: response.status,
            headers: Array.from(response.headers.entries())
          });
          return response;
        });
      };
    });
  }

  async stop() {
    this.recording = false;

    // 收集非确定性数据
    if (this.page) {
      const recordingData = await this.page.evaluate(() => {
        return (window as any).__recordingData;
      });

      // 转换为标准格式
      recordingData.random.forEach((item: any, index: number) => {
        this.nonDeterministicData.push({
          type: 'random',
          executionPoint: index,
          data: item
        });
      });

      recordingData.time.forEach((item: any, index: number) => {
        this.nonDeterministicData.push({
          type: 'time',
          executionPoint: index,
          data: item
        });
      });

      recordingData.network.forEach((item: any, index: number) => {
        this.nonDeterministicData.push({
          type: 'network',
          executionPoint: index,
          data: item
        });
      });
    }

    // 保存录制数据
    this.save();

    // 关闭浏览器
    if (this.browser) {
      await this.browser.close();
    }
  }

  private save() {
    if (!this.outputFile) return;

    const data = {
      version: '1.0',
      timestamp: Date.now(),
      executionPoints: this.executionPoints,
      nonDeterministicData: this.nonDeterministicData
    };

    fs.writeFileSync(this.outputFile, JSON.stringify(data, null, 2));
    console.log(`Recording saved to ${this.outputFile}`);
    console.log(`Total execution points: ${this.executionPoints.length}`);
    console.log(`Non-deterministic data: ${this.nonDeterministicData.length}`);
  }
}
```

## 方案 3: 最简化方案（快速原型）

如果你只想快速验证概念，可以用纯 JavaScript 实现：

### 3.1 简单的函数调用追踪

```javascript
// simple-recorder.js
class SimpleRecorder {
  constructor() {
    this.recording = false;
    this.executionLog = [];
    this.callDepth = 0;
  }

  start() {
    this.recording = true;
    this.executionLog = [];
    this.callDepth = 0;
    console.log('[Recorder] Started');
  }

  stop() {
    this.recording = false;
    console.log('[Recorder] Stopped');
    return this.executionLog;
  }

  log(type, data) {
    if (!this.recording) return;

    this.executionLog.push({
      id: this.executionLog.length,
      timestamp: Date.now(),
      type,
      depth: this.callDepth,
      ...data
    });
  }

  // 包装函数以追踪调用
  wrapFunction(fn, name) {
    const recorder = this;

    return function(...args) {
      recorder.callDepth++;
      recorder.log('call', {
        function: name,
        args: args.map(a => JSON.stringify(a))
      });

      let result;
      try {
        result = fn.apply(this, args);
        recorder.log('return', {
          function: name,
          result: JSON.stringify(result)
        });
      } catch (error) {
        recorder.log('error', {
          function: name,
          error: error.message
        });
        throw error;
      } finally {
        recorder.callDepth--;
      }

      return result;
    };
  }

  // 拦截非确定性函数
  interceptNonDeterministic() {
    const recorder = this;

    // Math.random
    const originalRandom = Math.random;
    Math.random = function() {
      const value = originalRandom();
      recorder.log('random', { value });
      return value;
    };

    // Date.now
    const originalDateNow = Date.now;
    Date.now = function() {
      const value = originalDateNow();
      recorder.log('time', { value });
      return value;
    };
  }

  save(filename) {
    const fs = require('fs');
    fs.writeFileSync(filename, JSON.stringify(this.executionLog, null, 2));
    console.log(`[Recorder] Saved to ${filename}`);
  }
}

// 使用示例
const recorder = new SimpleRecorder();
recorder.interceptNonDeterministic();

// 包装要追踪的函数
function fibonacci(n) {
  if (n <= 1) return n;
  return fibonacci(n - 1) + fibonacci(n - 2);
}

const trackedFib = recorder.wrapFunction(fibonacci, 'fibonacci');

// 开始录制
recorder.start();
```
